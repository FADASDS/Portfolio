217 public static boolean checkData(String typeOfConsumable, String nameOfConsumable, LocalDate
218 startData, LocalDate endData) throws Exception {
219
220 if (typeOfConsumable == null || typeOfConsumable.length() == 0) {
221 throw new Exception(ExceptionStrings.emptyTypeOfConsumable);
222 }
223 if (nameOfConsumable == null || nameOfConsumable.length() == 0) {
224 throw new Exception(ExceptionStrings.emptyNameOfConsumable);
225 }
226 if (startData == null) {
227 throw new Exception(ExceptionStrings.emptyStartData);
228 }
229 if (endData == null) {
230 throw new Exception(ExceptionStrings.emptyEndData);
231 }
232 if (endData.isBefore(startData)) {
233 throw new Exception(ExceptionStrings.wrongDate);
234 }
235
236 return true;
237 }

239 private void showDatePickerDialog(EditText field) {

251 String selectedDate = dayOfMonth + "/" + (month + 1) + "/" + year;
252 field.setText(selectedDate);
253 }
254 }, year, month, dayOfMonth);
